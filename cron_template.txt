# Finploy Sitemap Generator Cron Jobs
# Add these entries to your crontab using: crontab -e

# Daily sitemap generation at 2:00 AM
0 2 * * * /opt/finploy-sitemap-generator/run_sitemap_generator.sh --clean --log-level INFO >/dev/null 2>&1

# Weekly deep crawl on Sundays at 3:00 AM
0 3 * * 0 /opt/finploy-sitemap-generator/run_sitemap_generator.sh --clean --max-depth 6 --log-level INFO >/dev/null 2>&1

# Hourly quick validation (no crawling, just validate existing sitemaps)
0 * * * * /opt/finploy-sitemap-generator/run_sitemap_generator.sh --validate-only --log-level WARNING >/dev/null 2>&1

# Example with custom configuration
# 0 4 * * * /opt/finploy-sitemap-generator/run_sitemap_generator.sh --max-concurrent 5 --crawl-delay 2.0 --clean >/dev/null 2>&1
